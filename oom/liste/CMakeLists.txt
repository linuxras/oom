##
## Expand Qt macros in source files
##
QT5_WRAP_CPP ( liste_mocs
      # listedit.h
      # ctrllistedit.h
      # tracklistedit.h
      # partlistedit.h
      # ieventdialog.h
      editevent.h
      listedit.h
      )

##
## UI files
##
file (GLOB liste_ui_files
      editctrlbase.ui
      )
QT5_WRAP_UI ( liste_uis ${liste_ui_files} )

##
## List of source files to compile
##
file (GLOB liste_source_files
      # listedit.cpp
      # ctrllistedit.cpp
      # partlistedit.cpp
      # tracklistedit.cpp
      # ieventdialog.cpp
      editevent.cpp
      listedit.cpp
      )

##
## Define target
##
add_library ( liste STATIC
      ${liste_source_files}
      ${liste_mocs}
      ${liste_uis}
      )

##
## Append to the list of translations
##
set (FILES_TO_TRANSLATE
      ${FILES_TO_TRANSLATE}
      ${liste_source_files}
      ${liste_ui_files}
      CACHE INTERNAL ""
      )

##
## Compilation flags and target name
##
set_target_properties( liste
      PROPERTIES COMPILE_FLAGS "-include ${PROJECT_BINARY_DIR}/all.h -fPIC"
      OUTPUT_NAME oom_liste
      )

##
## Linkage
##
target_link_libraries ( liste
      ${Qt5Core_LIBRARIES}
      ${Qt5Widgets_LIBRARIES}
      ${Qt5Network_LIBRARIES}
      ${Qt5Concurrent_LIBRARIES}
      ${Qt5Svg_LIBRARIES}
      ${Qt5Xml_LIBRARIES}
      ${QT_QTXML_LIBRARY}
      awl
      widgets
      )
