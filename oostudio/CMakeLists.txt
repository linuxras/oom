##
## Expand Qt macros in source files
##
QT5_WRAP_CPP ( studio_mocs
	OOProcess.h
	OOStudio.h
	OOClient.h
	OODownload.h
	OOExtract.h
	OOThread.h
      )

##
## Resource files
##
QT5_ADD_RESOURCES (studio_qrc_files
		  studio.qrc
		  )

##
## List of source files to compile
##
file (GLOB studio_source_files
	OOProcess.cpp
    OOStudio.cpp
	OOClient.cpp
	OODownload.cpp
	OOExtract.cpp
	OOThread.cpp
	main.cpp
    )

file (GLOB studio_source_files
	${studio_source_files}
)

##
## UI files
##
file (GLOB studio_ui_files
	OOStudioBase.ui
	)

QT5_WRAP_UI (studio_ui_headers ${studio_ui_files})


##
## Define target
##
add_executable ( oostudio
      ${studio_ui_headers}
      ${studio_source_files}
      ${studio_mocs}
      ${studio_qrc_files}
      )

##
## Append to the list of translations
##
set (FILES_TO_TRANSLATE
      ${FILES_TO_TRANSLATE}
      ${studio_source_files}
      ${studio_ui_files}
      CACHE INTERNAL ""
      )

##
## Linkage
##
target_link_libraries ( oostudio
      ${Qt5Core_LIBRARIES}
      ${Qt5Widgets_LIBRARIES}
      ${Qt5Network_LIBRARIES}
      ${Qt5Concurrent_LIBRARIES}
      ${Qt5Svg_LIBRARIES}
      ${Qt5Xml_LIBRARIES}
      ${QT_QTXML_LIBRARY}
	  ${LSCP_LIBRARIES}
      ${JACK_LIBRARIES}
      )


##
## Install location
##
install(TARGETS oostudio
      DESTINATION ${CMAKE_INSTALL_PREFIX}/bin
      )
